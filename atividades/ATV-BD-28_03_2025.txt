CREATE VIEW EX1 AS 
SELECT N.*, C.area, C.cod_area, C.quadro, C.bonus, C.contratacao,CL.*,CP.estado, CP.regiao
FROM nivel N
INNER JOIN cargo C ON N.cod_nivel = C.cod_nivel
INNER JOIN cliente CL ON C.cod_cargo = CL.cod_cargo_responsavel
INNER JOIN cep CP ON CL.cep = CP.cep;
 
CREATE VIEW EX2 AS
SELECT CP.regiao, COUNT(C.cod_cliente) AS Quantidade
FROM cliente C
INNER JOIN CEP CP ON C.cep = CP.cep
GROUP BY CP.regiao;
 
CREATE VIEW EX3 AS 
SELECT N.descricao_nivel, SUM(F.salario_base + F.beneficios + F.vt + F.vr +F.impostos) AS SALARIO_TOTAL
FROM nivel N 
INNER JOIN cargo C ON N.cod_nivel = C.cod_nivel
INNER JOIN funcionario F ON C.cod_cargo = F.cod_cargo
GROUP BY N.descricao_nivel;

CREATE VIEW EX04A AS
SELECT CE.ESTADO, COUNT(F.COD_RH) AS QUANTIDADE_FUNCIONARIOS  
FROM FUNCIONARIO F INNER JOIN
CEP CE ON CE.CEP = F.CEP GROUP BY ESTADO

CREATE VIEW EX04B AS
SELECT CA.CONTRATACAO, COUNT(F.COD_RH) AS QUANTIDADE_CLIENTE 
FROM FUNCIONARIO F INNER JOIN CARGO CA ON F.COD_CARGO = CA.COD_CARGO
GROUP BY CA.CONTRATACAO

CREATE VIEW EX05A
SELECT CE.REGIAO, COUNT(C.COD_CLIENTE) AS QUANTIDADE_CLIENTE
FROM CLIENTE C INNER JOIN CEP CE ON C.CEP = CE.CEP
GROUP BY CE.REGIAO


CREATE VIEW EX05B AS
SELECT CA.AREA, COUNT(C.COD_CLIENTE) AS QUANTIDADE_FUNCIONARIOS  
FROM CLIENTE C INNER JOIN
CARGO CA ON C.COD_CARGO_RESPONSAVEL = CA.COD_CARGO GROUP BY AREA

CREATE VIEW EX05C AS
SELECT N.DESCRICAO_NIVEL, COUNT(C.COD_CLIENTE) AS QUANTIDADE_FUNCIONARIOS  
FROM CLIENTE C INNER JOIN
CARGO CA ON C.COD_CARGO_RESPONSAVEL = CA.COD_CARGO 
INNER JOIN NIVEL N ON CA.COD_NIVEL = N.COD_NIVEL
GROUP BY N.DESCRICAO_NIVEL

CREATE VIEW EX06 AS
SELECT N.DESCRICAO_NIVEL, SUM(C.VALOR_CONTRATO_ANUAL) AS QUANTIDADE_FUNCIONARIOS  
FROM CLIENTE C INNER JOIN
CARGO CA ON C.COD_CARGO_RESPONSAVEL = CA.COD_CARGO 
INNER JOIN NIVEL N ON CA.COD_NIVEL = N.COD_NIVEL
GROUP BY N.DESCRICAO_NIVEL

CREATE VIEW EX08A AS
SELECT CA.COD_CARGO, SUM(F.salario_base) AS TOTAL_SALARIO_BASE, 
SUM(F.SALARIO_BASE+F.IMPOSTOS+F.BENEFICIOS+F.VT+F.VR) AS TOTAL_SALARIO_BASE,
SUM(F.dias_uteis_trabalhados_ano_orcamentario) AS TOTAL_DIAS
FROM FUNCIONARIO F INNER JOIN CARGO CA ON F.COD_CARGO = CA.COD_CARGO GROUP BY CA.COD_CARGO

CREATE VIEW EX08B AS
SELECT CE.REGIAO, SUM(F.salario_base) AS TOTAL_SALARIO_BASE, 
SUM(F.SALARIO_BASE+F.IMPOSTOS+F.BENEFICIOS+F.VT+F.VR) AS TOTAL_SALARIO_BASE,
SUM(F.dias_uteis_trabalhados_ano_orcamentario) AS TOTAL_DIAS FROM FUNCIONARIO F
INNER JOIN CEP CE ON F.CEP = CE.CEP GROUP BY CE.REGIAO

CREATE VIEW EX08C AS
SELECT N.DESCRICAO_NIVEL, SUM(F.salario_base) AS TOTAL_SALARIO_BASE, 
SUM(F.SALARIO_BASE+F.IMPOSTOS+F.BENEFICIOS+F.VT+F.VR) AS TOTAL_SALARIO_BASE,
SUM(F.dias_uteis_trabalhados_ano_orcamentario) AS TOTAL_DIAS FROM FUNCIONARIO F
INNER JOIN CARGO CA ON F.COD_CARGO = CA.COD_CARGO INNER JOIN NIVEL N ON CA.COD_NIVEL = N.COD_NIVEL
GROUP BY N.DESCRICAO_NIVEL
